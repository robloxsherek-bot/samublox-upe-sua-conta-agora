        /* --- Funções do TOAST (Notificação) --- */

        /**
         * Exibe uma notificação customizada no topo da tela.
         * @param {string} message - A mensagem a ser exibida.
         */
        function showToast(message) {
            // 1. Define a mensagem
            toastTextElement.textContent = message;

            // 2. Mostra o toast
            toastElement.classList.remove('opacity-0', 'pointer-events-none');
            toastElement.classList.add('opacity-100');

            // 3. Esconde o toast após 3 segundos
            setTimeout(() => {
                toastElement.classList.remove('opacity-100');
                toastElement.classList.add('opacity-0', 'pointer-events-none');
            }, 3000); 
        }

        /* --- Funções do CHAT --- */

        /**
         * Adiciona uma nova mensagem ao chat.
         * @param {string} text - O texto da mensagem (pode ser HTML para fontes/tabela).
         * @param {string} sender - 'samublox' ou 'user'.
         */
        function addMessage(text, sender) {
            const messageDiv = document.createElement('div');
            // Alinha a mensagem do usuário à direita
            messageDiv.className = sender === 'samublox' ? 'message-samublox' : 'message-user ml-auto';
            messageDiv.innerHTML = text;
            chatMessages.appendChild(messageDiv);
            
            // Rola automaticamente para a última mensagem
            chatMessages.scrollTop = chatMessages.scrollHeight;
        }

        /**
         * Fecha o modal de Chat.
         */
        function closeChatModal() {
            // Anima o conteúdo
            chatModal.querySelector('.chat-modal-content').style.transform = 'scale(0.95)';
            setTimeout(() => {
                chatModal.classList.remove('modal-active');
                chatModal.classList.add('modal-inactive');
            }, 300);
        }
        
        /**
         * Adiciona o conteúdo (tabela/informação) e a opção de fechar o chat.
         * @param {string} contentHTML - O conteúdo a ser exibido (tabela, etc.).
         */
        function displayContentAndCloseOption(contentHTML) {
            // 1. Adiciona o conteúdo (Tabela/Info)
            addMessage(contentHTML, 'samublox');

            // 2. Adiciona o botão de fechar
            chatInputArea.innerHTML = '';
            const closeButton = document.createElement('button');
            closeButton.textContent = 'Fechar Chat';
            closeButton.className = 'w-full main-btn py-3 mt-4 rounded-md uppercase';
            closeButton.onclick = closeChatModal;
            chatInputArea.appendChild(closeButton);
        }

        /**
         * Exibe a mensagem de encerramento e o botão "Mudei de Ideia".
         */
        function displayClosingMessage(gameName) {
            const closingMessage = "Sem problemas! Se mudar de ideia ou precisar de um serviço muito específico, não hesite em nos chamar no Instagram: <a href='https://www.instagram.com/sam_ublox' target='_blank' class='text-[#FFFF00] underline font-bold'>@sam_ublox</a>. Tenha um ótimo dia!";
            addMessage(closingMessage, 'samublox');
            
            // Adiciona o botão de "Mudei de Ideia" e de Fechar
            chatInputArea.innerHTML = `
                <div class="flex flex-col space-y-2">
                    <button class="w-full chat-option-btn-second-chance py-3 rounded-md uppercase" onclick="handleAcceptance('${gameName}', true)">
                        Espere! Mudei de ideia, aceito a tabela
                    </button>
                    <button class="w-full main-btn py-3 rounded-md uppercase" onclick="closeChatModal()">
                        Fechar Chat
                    </button>
                </div>
            `;
            // Rola para o final
            chatMessages.scrollTop = chatMessages.scrollHeight;
        }

        /**
         * Exibe as opções de Sim/Não após a saudação.
         * @param {string} gameName - O nome do jogo.
         */
        function showAcceptanceOptions(gameName) {
            let acceptanceMessage = "";

            if (gameName === 'Geral') {
                 // Para suporte geral, não precisa de "Aceito/Não aceito a tabela", apenas mostra a info
                 const initialMessage = "Você clicou em **Suporte Geral**. Aqui estão as informações de contato e a forma de atendimento:";
                 addMessage(initialMessage, 'samublox');
                 
                 // Simula um pequeno delay antes de mostrar o conteúdo geral
                 setTimeout(() => {
                    const tableHTML = GAME_PRICES_MAP['Geral'];
                    displayContentAndCloseOption(tableHTML);
                 }, 500); 

            } else {
                // Para contratação de jogos
                acceptanceMessage = "Para prosseguir com o seu interesse em **" + gameName + "**, você aceita visualizar a tabela de preços e informações do serviço?";
                addMessage(acceptanceMessage, 'samublox');

                chatInputArea.innerHTML = `
                    <div class="flex space-x-2">
                        <button class="flex-1 chat-option-btn-accept py-3 rounded-md uppercase" onclick="handleAcceptance('${gameName}', true)">
                            Aceito a Tabela
                        </button>
                        <button class="flex-1 chat-option-btn-deny py-3 rounded-md uppercase" onclick="handleAcceptance('${gameName}', false)">
                            Não, obrigado
                        </button>
                    </div>
                `;
                // Garante que a área de opções fique visível
                chatMessages.scrollTop = chatMessages.scrollHeight;
            }
        }

        /**
         * Lida com a resposta do usuário (Aceitar ou Negar a tabela).
         * @param {string} gameName - O nome do jogo.
         * @param {boolean} accepted - True se aceitou, False se negou.
         */
        function handleAcceptance(gameName, accepted) {
            chatInputArea.innerHTML = ''; // Limpa as opções

            if (accepted) {
                addMessage('Aceito a Tabela de Preços!', 'user');
                
                const tableHTML = GAME_PRICES_MAP[gameName] || "Desculpe, a tabela de preços para este jogo não está disponível no momento.";
                
                // Pequeno delay para a mensagem de aceitação aparecer antes da tabela
                setTimeout(() => {
                    const followUpMessage = "Excelente! Aqui está a nossa tabela completa de preços e informações para **" + gameName + "**:";
                    addMessage(followUpMessage, 'samublox');
                    displayContentAndCloseOption(tableHTML);
                }, 500); 

            } else {
                addMessage('Não, obrigado.', 'user');
                
                setTimeout(() => {
                    displayClosingMessage(gameName); // Passa o nome do jogo para a opção "Mudei de Ideia"
                }, 500);
            }
        }

         * Lida com a abertura do chat.
         * @param {string} sourceName - O nome do jogo ou 'Geral'.

        function openChatModal(sourceName) {
            currentGameName = sourceName; 
            
            // 1. Abre o modal
            chatModal.classList.remove('modal-inactive');
            chatModal.classList.add('modal-active');
            // Anima o conteúdo
            chatModal.querySelector('.chat-modal-content').style.transform = 'scale(1)';
            
            // 2. Limpa o chat
            chatMessages.innerHTML = ''; 
            chatInputArea.innerHTML = '';
            
            chatTitle.textContent = sourceName === 'Geral' ? "SamuBlox - Suporte Geral" : `SamuBlox - Contratar ${sourceName}`;
            
            // MENSAGEM DE SAUDAÇÃO
            const initialMessage = "OLÁ! TUDO BEM!? Me chamo SamuBlox, eu sou o administrador do site e irei te atender!";
            addMessage(initialMessage, 'samublox');

            // Simula um pequeno delay (2 segundos) antes de mostrar as opções de aceitação/info
            setTimeout(() => {
                showAcceptanceOptions(sourceName);
            }, 2000); 
        }
